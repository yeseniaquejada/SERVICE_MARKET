/*CREACION DE DISPARADOR QUE BLOQUEE ACTUALIZAR EL ID DEL SERVICIO CUANDO SE REALICE LA ACTUALIZACION*/
CREATE TRIGGER TR_SERVICIO_ACTUALIZAR_ID
ON SERVICIOS
FOR UPDATE
AS
IF UPDATE(ID_SERVICIO)
BEGIN
PRINT 'NO SE PUEDE ACTUALIZAR EL ID DEL SERVICIO'
ROLLBACK TRANSACTION
END 

/*CREACION DE DISPARADOR QUE NO PERMITE ELIMINAR MÁS DE UN SEVRICIO*/
CREATE TRIGGER TR_SERVICIOS_BORRAR
ON SERVICIOS
FOR DELETE
AS
IF (SELECT COUNT(*) FROM DELETED) > 1
BEGIN
PRINT('NO PUEDE ELIMINAR MÁS DE UN SERVICIO')
ROLLBACK TRANSACTION
END
